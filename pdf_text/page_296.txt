# bin 에 속한 값들의 요약 통계 보여주는 함수 
fn_binvalues <- function(x, hist.object) { 
bins <- findlnterval(x, hist.object$breaks) 
} 
bins_factor <- factor(bins, levels = sort(unique(bins))) 
aggregate(x, by = list(bins_factor) , summary) 
fn_binvalues(dat_normal, hist_normal) 
fn_binvalues(dat_MCI, hist_MCI) 
fn_binvalues(dat_Dementia, hist_dementia)
