outlier_detecting2 <- function(x) { 
} 
Q1 <- quantile(x, 0.25, na.rm = TRUE) 
Q3 <- quantile(x, 0.75, na.rm = TRUE) 
IQR <- Q3 - Q1 
lower <- Q1 - 1.5 * IQR 
upper <- Q3 + 1.5 * IQR 
outliers_id <- which((x < lower)l(x > upper)) 
return(outliers_id)
