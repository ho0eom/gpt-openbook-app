my_mode3 <- function(dat) { 
p <- ncol(dat) 
} 
modes <- list() 
for (k in 1:p) { 
} 
x <- dat[,k] 
x <- na.omit(x) 
x_unique <- unique(x) 
x_len <- c() 
for (i in 1 :length(x_unique)) x_len[i] <- sum(x == x_unique[i]) 
modes[[k]] <- x_unique[x_len == max(x_len)] 
names(modes) <- colnames(dat) 
return(modes)
