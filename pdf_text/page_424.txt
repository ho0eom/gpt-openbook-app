# 분산, 표준펀자, 변동계수 (방법 1) 
scores <- dementia[c("ADAS11 ", "ADAS1 3", "MMSE")] 
xnames <- colnames(scores) 
res <- matrix(NA, nrow = 3, ncol = 3) 
for (i in 1:3) { 
xi <- scores[, i] 
res[, i] <- c(var(xi, na.rm = TRUE), sd(xi, na.rm = TRUE), my_CV(xi, 
na.rm = TRUE)) 
} 
colnames(res) <- xnames 
rownames(res) <- c("var" , "sd" , "cv") 
round(res, 3)
